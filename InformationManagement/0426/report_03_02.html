def calc_cross_corr(x, y):

  num_samples = np.size(y)
  cross_corr = np.zeros(num_samples)

  for m in range(num_samples):
    shifted_y = np.zeros(num_samples)
    shifted_y[:num_samples-m] = y[m:]
    cross_corr[m] = np.dot(x, shifted_y)
    
  return cross_corr
